Actually — correction: In Next.js App Router, client components CAN render server components only
if they are passed as children/props. Since these project pages ARE the page.tsx (route entry point)
and they have `'use client'`, the simplest approach is:
- For the 6 client-side pages: Add `<script type="application/ld+json">` directly in the JSX

  pattern already used in these pages' layout.

  This is more complex and not worth it for this phase.
- Best approach: Just render the JSON-LD inline in the client component. The `safeJsonLdStringify`
  utility works in both client and server components. The data is all hardcoded (no server-only deps).

    ```typescript
    import { safeJsonLdStringify } from '@/lib/json-ld-utils'

    export function BreadcrumbListJsonLd({
      items,
      nonce,
    }: {
      items: Array<{ name: string; url: string }>
      nonce?: string | null
    }) {
      const jsonLd = {
        '@context': 'https://schema.org',
        '@type': 'BreadcrumbList',
        itemListElement: items.map((crumb, index) => ({
          '@type': 'ListItem',
          position: index + 1,
          name: crumb.name,
          item: crumb.url,
        })),
      }

      return (
        <script
          type="application/ld+json"
          nonce={nonce ?? undefined}
          dangerouslySetInnerHTML={{ __html: safeJsonLdStringify(jsonLd) }}
        />
      )
    }
    ```

    The `items` prop uses `url` (not `item`) for clarity — the schema.org property `item` maps to the URL. Use full production URLs: `https://richardwhudsonjr.com`, `https://richardwhudsonjr.com/projects`, `https://richardwhudsonjr.com/projects/slug`.

    ```typescript
    export function FAQPageJsonLd({
      faqs,
      nonce,
    }: {
      faqs: Array<{ question: string; answer: string }>
      nonce?: string | null
    }) {
      const jsonLd = {
        '@context': 'https://schema.org',
        '@type': 'FAQPage',
        mainEntity: faqs.map((faq) => ({
          '@type': 'Question',
          name: faq.question,
          acceptedAnswer: {
            '@type': 'Answer',
            text: faq.answer,
          },
        })),
      }

      return (
        <script
          type="application/ld+json"
          nonce={nonce ?? undefined}
          dangerouslySetInnerHTML={{ __html: safeJsonLdStringify(jsonLd) }}
        />
      )
    }
    ```
